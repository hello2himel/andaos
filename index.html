    Install
    Flash → Boot → pacstrap. Standard Arch — nothing proprietary.
    sudo dd bs=4M if=andaos-*.iso of=/dev/sdX \ status=progress oflag=sync # then pacstrap /mnt base linux hyprland networkmanager genfstab -U /mnt >> /mnt/etc/fstab
